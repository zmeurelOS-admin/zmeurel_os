1	import { redirect } from 'next/navigation'
2	
3	interface CallbackPageProps {
4	  searchParams?: Promise<Record<string, string | string[] | undefined>>
5	}
6	
7	function buildQueryString(params: Record<string, string | string[] | undefined>): string {
8	  const query = new URLSearchParams()
9	
10	  Object.entries(params).forEach(([key, rawValue]) => {
11	    if (Array.isArray(rawValue)) {
12	      rawValue.forEach((value) => query.append(key, value))
13	      return
14	    }
15	    if (typeof rawValue === 'string') {
16	      query.set(key, rawValue)
17	    }
18	  })
19	
20	  const serialized = query.toString()
21	  return serialized ? `?${serialized}` : ''
22	}
23	
24	export default async function CallbackPage({ searchParams }: CallbackPageProps) {
25	  const resolvedSearchParams = searchParams ? await searchParams : {}
26	  const queryString = buildQueryString(resolvedSearchParams)
27	
28	  redirect(`/auth/callback${queryString}`)
29	}
