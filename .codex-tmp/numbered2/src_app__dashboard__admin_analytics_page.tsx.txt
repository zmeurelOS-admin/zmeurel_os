1	import { redirect } from 'next/navigation'
2	import { AnalyticsDashboard } from '@/components/admin/AnalyticsDashboard'
3	import { createClient } from '@/lib/supabase/server'
4	import { isSuperAdmin } from '@/lib/auth/isSuperAdmin'
5	
6	export default async function AdminAnalyticsPage() {
7	  const supabase = await createClient()
8	  const {
9	    data: { user },
10	  } = await supabase.auth.getUser()
11	
12	  if (!user) {
13	    redirect('/login')
14	  }
15	
16	  const superadmin = await isSuperAdmin(supabase, user.id)
17	  if (!superadmin) {
18	    redirect('/dashboard')
19	  }
20	
21	  return <AnalyticsDashboard />
22	}
