1	'use client'
2	
3	import { useEffect } from 'react'
4	
5	export function ServiceWorkerRegister() {
6	  useEffect(() => {
7	    if (process.env.NODE_ENV !== 'production') return
8	    if (typeof window === 'undefined' || !('serviceWorker' in navigator)) return
9	
10	    const isSecure = window.location.protocol === 'https:' || window.location.hostname === 'localhost'
11	    if (!isSecure) return
12	
13	    const register = async () => {
14	      try {
15	        await navigator.serviceWorker.register('/sw.js', { scope: '/' })
16	      } catch {
17	        // no-op
18	      }
19	    }
20	
21	    register()
22	  }, [])
23	
24	  return null
25	}
