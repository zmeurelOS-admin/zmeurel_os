  12: export async function getParcele(): Promise<Parcela[]> {
  13:   const supabase = getSupabase()
  14: 
  15:   const { data, error } = await supabase
  16:     .from("parcele")
  17:     .select("id,id_parcela,nume_parcela,tip_fruct,soi_plantat,suprafata_m2,nr_plante,an_plantare,status,gps_lat,gps_lng,observatii,created_at,updated_at,tenant_id")
  18:     .order("created_at", { ascending: false })

  29: export async function createParcela(
  30:   input: ParcelaInsert
  31: ): Promise<Parcela> {
  32:   const supabase = getSupabase()
  33: 
  34:   const {
  35:     data: { user },
  36:     error: authError,
  37:   } = await supabase.auth.getUser()
  38: 
  39:   if (authError || !user) {
  40:     throw new Error("Neautorizat. Trebuie sa fii logat.")
  41:   }
  42: 
  43:   const { data: tenant, error: tenantError } = await supabase
  44:     .from("tenants")
  45:     .select("id")
  46:     .eq("owner_user_id", user.id)
  47:     .maybeSingle()
  48: 
  49:   if (tenantError) throw tenantError
  50:   if (!tenant?.id) throw new Error("Tenant indisponibil pentru utilizatorul curent.")
  51: 
  52:   const { data, error } = await supabase
  53:     .from("parcele")
  54:     .insert({
  55:       ...input,
  56:       tenant_id: tenant.id,
  57:     })
  58:     .select("id,id_parcela,nume_parcela,tip_fruct,soi_plantat,suprafata_m2,nr_plante,an_plantare,status,gps_lat,gps_lng,observatii,created_at,updated_at,tenant_id")
  59:     .single()
  60: 
  61:   if (error) throw error
  62: 
  63:   return data
  64: }
