41	// ========================================
42	// GENERATE NEXT ID
43	// ========================================
44	
45	async function generateNextId(
46	  supabase: ReturnType<typeof getSupabase>
47	): Promise<string> {
48	  const { data, error } = await supabase
49	    .from('culegatori')
50	    .select('id_culegator')
51	    .order('created_at', { ascending: true })
52	
53	  if (error) throw error
54	
55	  const ids = (data ?? [])
56	    .map((r) => r.id_culegator)
57	    .filter((id) => id && /^CUL\d+$/.test(id))
58	    .map((id) => parseInt(id!.substring(3)))
59	
60	  const maxId = ids.length > 0 ? Math.max(...ids) : 0
61	
62	  return `CUL${String(maxId + 1).padStart(3, '0')}`
63	}

69	export async function getCulegatori(): Promise<Culegator[]> {
70	  const supabase = getSupabase()
71	
72	  const { data, error } = await supabase
73	    .from('culegatori')
74	    .select('id,id_culegator,nume_prenume,tarif_lei_kg,data_angajare,status_activ,telefon,tip_angajare,observatii,created_at,updated_at,tenant_id')
75	    .order('created_at', { ascending: false })
76	
77	  if (error) throw error
78	
79	  return (data ?? []) as unknown as Culegator[]
80	}

82	// ========================================
83	// CREATE (FIXED VERSION)
84	// ========================================
85	
86	export async function createCulegator(
87	  input: CreateCulegatorInput
88	): Promise<Culegator> {
89	  const supabase = getSupabase()
90	  const id_culegator = await generateNextId(supabase)
91	
92	  const payload = {
93	    id_culegator,
94	    nume_prenume: input.nume_prenume,
95	    tarif_lei_kg: input.tarif_lei_kg,
96	    data_angajare:
97	      input.data_angajare && input.data_angajare !== ''
98	        ? input.data_angajare
99	        : null,
100	    status_activ: input.status_activ ?? true,
101	    telefon:
102	      input.telefon && input.telefon !== '' ? input.telefon : null,
103	    tip_angajare:
104	      input.tip_angajare && input.tip_angajare !== ''
105	        ? input.tip_angajare
106	        : null,
107	    observatii:
108	      input.observatii && input.observatii !== ''
109	        ? input.observatii
110	        : null,
111	  }
112	
113	  const { data, error } = await supabase
114	    .from('culegatori')
115	    .insert(payload)
116	    .select()
117	    .single()
118	
119	  if (error) throw error
120	
121	  return data as unknown as Culegator
122	}

166	// DELETE
167	// ========================================
168	
169	export async function deleteCulegator(id: string): Promise<void> {
170	  const supabase = getSupabase()
171	
172	  // Check for related recoltari before deleting
173	  const { count: recoltariCount, error: recoltariError } = await supabase
174	    .from('recoltari')
175	    .select('*', { count: 'exact', head: true })
176	    .eq('culegator_id', id)
177	
178	  if (recoltariError) throw recoltariError
179	
180	  // If any related records exist, throw a descriptive error
181	  if ((recoltariCount ?? 0) > 0) {
182	    throw new Error(`Nu poți șterge. Are ${recoltariCount} recoltări asociate.`)
183	  }
184	
185	  // Only proceed with delete if count is 0
186	  const { error } = await supabase
187	    .from('culegatori')
188	    .delete()
189	    .eq('id', id)
190	
191	  if (error) throw error
192	}
