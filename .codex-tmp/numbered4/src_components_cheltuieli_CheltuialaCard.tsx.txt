1	'use client'
2	
3	import { Pencil, Trash2 } from 'lucide-react'
4	
5	import { CompactListCard } from '@/components/app/CompactListCard'
6	import type { Cheltuiala } from '@/lib/supabase/queries/cheltuieli'
7	import { isAutoManoperaCheltuiala } from '@/lib/supabase/queries/manopera-auto'
8	
9	interface CheltuialaCardProps {
10	  cheltuiala: Cheltuiala
11	  onView: (cheltuiala: Cheltuiala) => void
12	  onEdit: (cheltuiala: Cheltuiala) => void
13	  onDelete: (id: string, name: string) => void
14	}
15	
16	export function CheltuialaCard({ cheltuiala, onView, onEdit, onDelete }: CheltuialaCardProps) {
17	  const categorie = cheltuiala.categorie || 'Altele'
18	  const isAutoGenerated = isAutoManoperaCheltuiala(cheltuiala)
19	  const hasDescription = Boolean(cheltuiala.descriere?.trim())
20	  const formattedDate = new Date(cheltuiala.data).toLocaleDateString('ro-RO')
21	  const descriptionShort = (cheltuiala.descriere || '').slice(0, 30)
22	
23	  return (
24	    <>
25	      <div className="cursor-pointer px-4 py-2.5 transition-opacity active:opacity-95 sm:hidden" onClick={() => onView(cheltuiala)}>
26	        <p className="truncate text-base font-semibold text-[var(--agri-text)]">
27	          {formattedDate} · {categorie} · <span className="value-money-negative">{Number(cheltuiala.suma_lei).toFixed(2)} lei</span>
28	        </p>
29	        <div className="mt-1 flex items-center justify-between gap-2">
30	          <div className="min-w-0">
31	            {isAutoGenerated ? (
32	              <span className="inline-flex items-center rounded-full bg-blue-100 px-1.5 py-0.5 text-xs font-semibold text-blue-700">
33	                ⚡ Auto
34	              </span>
35	            ) : null}
36	            <p className={`truncate ${isAutoGenerated ? 'text-xs text-gray-400 italic' : 'text-sm text-[var(--agri-text-muted)]'}`}>
37	              {isAutoGenerated ? 'Generat automat din recoltări' : descriptionShort || '-'}
38	            </p>
39	          </div>
40	          <div className="flex items-center gap-1">
41	            <button
42	              type="button"
43	              aria-label="Editeaza cheltuiala"
44	              className="inline-flex h-11 w-11 items-center justify-center p-2 text-gray-400 transition-colors hover:text-blue-600"
45	              onClick={(event) => {
46	                event.stopPropagation()
47	                onEdit(cheltuiala)
48	              }}
49	            >
50	              <Pencil className="h-4 w-4" />
51	            </button>
52	            <button
53	              type="button"
54	              aria-label="Sterge cheltuiala"
55	              className="inline-flex h-11 w-11 items-center justify-center p-2 text-gray-400 transition-colors hover:text-red-600"
56	              onClick={(event) => {
57	                event.stopPropagation()
58	                onDelete(cheltuiala.id, `${categorie} - ${cheltuiala.suma_lei} lei`)
59	              }}
60	            >
61	              <Trash2 className="h-4 w-4" />
62	            </button>
63	          </div>
64	        </div>
65	      </div>
66	
67	      <div className="hidden sm:block">
68	        <CompactListCard
69	          title="Cheltuiala"
70	          leftRows={[
71	            {
72	              label: 'Categorie',
73	              value: (
74	                <span className="inline-flex items-center gap-2 font-semibold">
75	                  <span>{categorie}</span>
76	                  {isAutoGenerated ? (
77	                    <span className="inline-flex items-center rounded-full bg-blue-100 px-1.5 py-0.5 text-xs font-semibold text-blue-700">
78	                      ⚡ Auto
79	                    </span>
80	                  ) : null}
81	                </span>
82	              ),
83	              valueClassName: 'font-semibold',
84	            },
85	            { label: 'Data', value: formattedDate, valueClassName: 'text-sm text-[var(--agri-text-muted)] font-medium' },
86	          ]}
87	          rightRows={[
88	            { label: 'Suma', value: `-${Number(cheltuiala.suma_lei).toFixed(2)} lei`, emphasis: 'financial', valueClassName: 'value-money-negative' },
89	            ...(hasDescription || isAutoGenerated
90	              ? [
91	                  {
92	                    label: 'Descriere',
93	                    value: isAutoGenerated ? 'Generat automat din recoltări' : (cheltuiala.descriere as string),
94	                    allowWrap: true,
95	                    forceSingleLine: false,
96	                    valueClassName: isAutoGenerated ? 'text-xs text-gray-400 italic' : 'text-sm text-[var(--agri-text-muted)] font-medium line-clamp-2',
97	                  },
98	                ]
99	              : []),
100	          ]}
101	          onClick={() => onView(cheltuiala)}
102	          onEdit={() => onEdit(cheltuiala)}
103	          onDelete={() => onDelete(cheltuiala.id, `${categorie} - ${cheltuiala.suma_lei} lei`)}
104	        />
105	      </div>
106	    </>
107	  )
108	}
