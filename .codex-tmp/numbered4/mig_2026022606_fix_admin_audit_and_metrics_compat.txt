23	alter table public.tenant_metrics_daily
24	  add column if not exists total_revenue_lei numeric not null default 0,
25	  add column if not exists updated_at timestamptz not null default now();

27	-- Backfill only when a legacy total_revenue column exists
28	do $$
29	begin
30	  if exists (
31	    select 1
32	    from information_schema.columns
33	    where table_schema = 'public'
34	      and table_name = 'tenant_metrics_daily'
35	      and column_name = 'total_revenue'
36	  ) then
37	    execute '
38	      update public.tenant_metrics_daily
39	      set total_revenue_lei = coalesce(total_revenue, 0)
40	      where coalesce(total_revenue_lei, 0) = 0
41	        and coalesce(total_revenue, 0) <> 0
42	    ';
43	  end if;
44	end
45	$$;

73	-- -----------------------------------------------------------------------------
74	-- 2) Fix admin_list_audit_logs return shape to match app expectations
75	-- -----------------------------------------------------------------------------
76	
77	drop function if exists public.admin_list_audit_logs(integer, integer);
78	
79	create function public.admin_list_audit_logs(
80	  p_limit integer default 50,
81	  p_offset integer default 0
82	)
83	returns table (
84	  id uuid,
85	  created_at timestamptz,
86	  actor_email text,
87	  tenant_name text,
88	  old_plan text,
89	  new_plan text,
90	  action text
91	)
92	language plpgsql
93	stable
94	security definer
95	set search_path = public
96	as $$
97	begin
98	  if not public.is_superadmin(auth.uid()) then
99	    raise exception 'FORBIDDEN';
100	  end if;
101	
102	  return query
103	  select
104	    a.id,
105	    a.created_at,
106	    u.email::text as actor_email,
107	    t.nume_ferma as tenant_name,
108	    a.old_plan,
109	    a.new_plan,
110	    a.action
111	  from public.audit_logs a
112	  left join public.tenants t on t.id = a.target_tenant_id
113	  left join auth.users u on u.id = a.actor_user_id
114	  order by a.created_at desc
115	  limit greatest(coalesce(p_limit, 50), 1)
116	  offset greatest(coalesce(p_offset, 0), 0);
117	end;
118	$$;
119	
120	grant execute on function public.admin_list_audit_logs(integer, integer) to authenticated;
121	grant execute on function public.admin_list_audit_logs(integer, integer) to service_role;
