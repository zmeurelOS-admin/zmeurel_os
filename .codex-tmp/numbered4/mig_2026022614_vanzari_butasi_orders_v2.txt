56	    from public.vanzari_butasi_items it

60	create or replace function public.enforce_vanzari_butasi_items_tenant()
61	returns trigger
62	language plpgsql
63	security definer
64	as $$
65	declare
66	  order_tenant uuid;
67	begin
68	  select tenant_id
69	  into order_tenant
70	  from public.vanzari_butasi
71	  where id = new.comanda_id;
72	
73	  if order_tenant is null then
74	    raise exception 'Comanda invalida sau fara tenant';
75	  end if;
76	
77	  if new.tenant_id is null then
78	    new.tenant_id := order_tenant;
79	  end if;
80	
81	  if new.tenant_id <> order_tenant then
82	    raise exception 'tenant_id din item trebuie sa corespunda comenzii';
83	  end if;
84	
85	  return new;
86	end;
87	$$;
