7	create table if not exists public.miscari_stoc (
8	  id uuid primary key default gen_random_uuid(),
9	  tenant_id uuid not null references public.tenants(id) on delete cascade,
10	  locatie_id uuid not null references public.parcele(id) on delete cascade,
11	  produs text not null,
12	  calitate text not null,
13	  depozit text not null,
14	  tip_miscare text not null,
15	  cantitate_kg numeric not null,
16	  referinta_id uuid,
17	  data date not null default current_date,
18	  observatii text,
19	  created_at timestamptz not null default now(),
20	  constraint miscari_stoc_cantitate_non_negative check (cantitate_kg >= 0),
21	  constraint miscari_stoc_calitate_check check (calitate in ('cal1', 'cal2')),
22	  constraint miscari_stoc_depozit_check check (depozit in ('fresh', 'congelat', 'procesat')),
23	  constraint miscari_stoc_tip_miscare_check check (
24	    tip_miscare in (
25	      'recoltare',
26	      'vanzare',
27	      'consum',
28	      'oferit_gratuit',
29	      'procesare',
30	      'congelare',
31	      'pierdere',
32	      'ajustare'
33	    )
34	  )
35	);
