65	function escapeCsvValue(value: unknown): string {
66	  const raw = value === null || value === undefined ? '' : String(value)
67	  const escaped = raw.replace(/"/g, '""')
68	  return `"${escaped}"`
69	}

71	function rowsToCsv(rows: Record<string, unknown>[]): string {
72	  if (!rows.length) return ''
73	
74	  const headers = Array.from(new Set(rows.flatMap((row) => Object.keys(row))))
75	  const headerLine = headers.map(escapeCsvValue).join(',')
76	  const lines = rows.map((row) => headers.map((key) => escapeCsvValue(row[key])).join(','))
77	
78	  return [headerLine, ...lines].join('\n')
79	}

81	function downloadFile(content: string, fileName: string, mimeType: string) {
82	  const blob = new Blob([content], { type: mimeType })
83	  const url = URL.createObjectURL(blob)
84	  const anchor = document.createElement('a')
85	  anchor.href = url
86	  anchor.download = fileName
87	  anchor.click()
88	  URL.revokeObjectURL(url)
89	}
