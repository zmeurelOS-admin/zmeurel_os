169	create or replace function public.set_audit_fields_minimal()
170	returns trigger
171	language plpgsql
172	security definer
173	as $$
174	begin
175	  if tg_op = 'INSERT' then
176	    new.created_at := coalesce(new.created_at, now());
177	    new.updated_at := coalesce(new.updated_at, now());
178	    new.created_by := coalesce(auth.uid(), new.created_by);
179	    new.updated_by := coalesce(auth.uid(), new.updated_by, new.created_by);
180	  else
181	    new.created_at := old.created_at;
182	    new.created_by := old.created_by;
183	    new.updated_at := now();
184	    new.updated_by := coalesce(auth.uid(), old.updated_by, old.created_by);
185	  end if;
186	  return new;
187	end;
188	$$;

230	declare
231	  tbl text;
232	  tenant_check text := 'tenant_id = (select id from public.tenants where owner_user_id = auth.uid())';
233	begin
234	  foreach tbl in array array['recoltari', 'vanzari', 'activitati_agricole', 'clienti', 'parcele', 'cheltuieli_diverse'] loop
235	    if to_regclass('public.' || tbl) is null then
236	      continue;
237	    end if;
238	
239	    execute format('alter table public.%I enable row level security', tbl);
240	
241	    execute format('drop policy if exists %I on public.%I', tbl || '_tenant_select', tbl);
242	    execute format('create policy %I on public.%I for select using (%s)', tbl || '_tenant_select', tbl, tenant_check);
243	
244	    execute format('drop policy if exists %I on public.%I', tbl || '_tenant_insert', tbl);
245	    execute format('create policy %I on public.%I for insert with check (%s)', tbl || '_tenant_insert', tbl, tenant_check);
246	
247	    execute format('drop policy if exists %I on public.%I', tbl || '_tenant_update', tbl);
248	    execute format('create policy %I on public.%I for update using (%s) with check (%s)', tbl || '_tenant_update', tbl, tenant_check, tenant_check);
249	  end loop;
250	end
