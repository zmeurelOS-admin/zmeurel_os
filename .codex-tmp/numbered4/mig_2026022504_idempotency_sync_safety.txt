50	-- ===============================
51	-- Audit + sync safety trigger
52	-- ===============================
53	
54	create or replace function public.set_sync_audit_fields()
55	returns trigger
56	language plpgsql
57	security definer
58	as $$
59	begin
60	  if new.client_sync_id is null then
61	    new.client_sync_id := gen_random_uuid();
62	  end if;
63	
64	  if new.sync_status is null then
65	    new.sync_status := 'synced';
66	  end if;
67	
68	  if tg_op = 'INSERT' then
69	    if new.created_by is null then
70	      new.created_by := auth.uid();
71	    end if;
72	    if new.updated_by is null then
73	      new.updated_by := coalesce(auth.uid(), new.created_by);
74	    end if;
75	  else
76	    new.updated_by := coalesce(auth.uid(), new.updated_by);
77	  end if;
78	
79	  new.updated_at := now();
80	  return new;
81	end;
82	$$;

165	-- ===============================
166	-- Generic idempotent upsert helper
167	-- ===============================
168	
169	create or replace function public.upsert_with_idempotency(table_name text, payload jsonb)
170	returns jsonb
171	language plpgsql
172	security definer
173	as $$
174	declare
175	  result jsonb;
176	begin
177	  if table_name not in ('recoltari', 'vanzari', 'activitati_agricole', 'cheltuieli_diverse') then
178	    raise exception 'Unsupported table: %', table_name;
179	  end if;
180	
181	  execute format(
182	    'with upserted as (
183	      insert into public.%1$I as t
184	      select * from jsonb_populate_record(null::public.%1$I, $1)
185	      on conflict (client_sync_id)
186	      do update
187	      set sync_status = coalesce(excluded.sync_status, t.sync_status),
188	          updated_by = coalesce(excluded.updated_by, auth.uid(), t.updated_by),
189	          updated_at = now()
190	      returning t.*
191	    )
192	    select to_jsonb(upserted) from upserted',
193	    table_name
194	  )
195	  into result
196	  using payload;
197	
198	  return result;
199	end;
200	$$;
