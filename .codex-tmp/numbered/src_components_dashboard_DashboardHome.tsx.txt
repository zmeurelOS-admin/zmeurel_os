101	function getTodayDateString(): string {
102	  const today = new Date();
103	  const year = today.getFullYear();
104	  const month = String(today.getMonth() + 1).padStart(2, '0');
105	  const day = String(today.getDate()).padStart(2, '0');
106	  return `${year}-${month}-${day}`;
107	}

109	function useKPIs() {
110	  const [kpis, setKpis] = useState<KPIData>({
111	    totalRecoltatAzi: 0,
112	    totalVanzariAzi: 0,
113	  });
114	  const [loading, setLoading] = useState(true);
115	
116	  useEffect(() => {
117	    async function fetchKPIs() {
118	      try {
119	        const today = getTodayDateString();
120	        const [recoltari, vanzari] = await Promise.all([
121	          getRecoltari(),
122	          getVanzari(),
123	        ]);
124	
125	        const totalRecoltatAzi = recoltari
126	          .filter((r) => r.data === today)
127	          .reduce((sum, r) => sum + Number(r.kg_cal1 || 0) + Number(r.kg_cal2 || 0), 0);
128	
129	        const totalVanzariAzi = vanzari
130	          .filter((v) => v.data === today)
131	          .reduce((sum, v) => sum + v.cantitate_kg * v.pret_lei_kg, 0);
132	
133	        setKpis({ totalRecoltatAzi, totalVanzariAzi });
134	      } catch (error) {
135	        console.error('Error fetching KPIs:', error);
136	      } finally {
137	        setLoading(false);
138	      }
139	    }
140	
141	    fetchKPIs();
142	  }, []);
143	
144	  return { kpis, loading };
145	}

147	function useActivityFeed() {
148	  const [activities, setActivities] = useState<ActivityItem[]>([]);
149	  const [loading, setLoading] = useState(true);
150	
151	  useEffect(() => {
152	    async function fetchActivities() {
153	      try {
154	        const [recoltari, vanzari, activitati] = await Promise.all([
155	          getRecoltari(),
156	          getVanzari(),
157	          getActivitatiAgricole(),
158	        ]);
159	
160	        const normalized: ActivityItem[] = [];
161	
162	        recoltari.slice(0, 5).forEach((r) => {
163	          const totalKg = Number(r.kg_cal1 || 0) + Number(r.kg_cal2 || 0);
164	          normalized.push({
165	            id: r.id,
166	            type: 'recoltare',
167	            title: 'Recoltare',
168	            subtitle: `${totalKg} kg`,
169	            value: `${totalKg} kg`,
170	            date: r.data,
171	          });
172	        });
173	
174	        vanzari.slice(0, 5).forEach((v) => {
175	          normalized.push({
176	            id: v.id,
177	            type: 'vanzare',
178	            title: 'VДӮЛҳnzare',
179	            subtitle: `${v.cantitate_kg} kg ГӮВ· ${v.pret_lei_kg} lei/kg`,
180	            value: `${(v.cantitate_kg * v.pret_lei_kg).toFixed(2)} lei`,
181	            date: v.data,
182	          });
183	        });
184	
185	        activitati.slice(0, 5).forEach((a) => {
186	          normalized.push({
187	            id: a.id,
188	            type: 'activitate',
189	            title: a.tip_activitate || 'Activitate agricolГ„Вғ',
190	            subtitle: a.produs_utilizat || 'FГ„ВғrГ„Вғ produs specificat',
191	            date: a.data_aplicare,
192	          });
193	        });
194	
195	        normalized.sort((a, b) => new Date(b.date).getTime() - new Date(a.date).getTime());
196	
197	        setActivities(normalized);
198	      } catch (error) {
199	        console.error('Error fetching activities:', error);
200	      } finally {
201	        setLoading(false);
202	      }
203	    }
204	
205	    fetchActivities();
206	  }, []);
207	
208	  return { activities, loading };
209	}
