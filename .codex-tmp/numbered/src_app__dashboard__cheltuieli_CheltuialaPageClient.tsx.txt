51	function isSchemaCacheError(error: unknown): boolean {
52	  const message = String((error as { message?: string })?.message ?? '').toLowerCase()
53	  return message.includes('schema cache') || message.includes('could not find the')
54	}
55	
56	function handleCheltuialaError(error: unknown, fallbackMessage: string) {
57	  const message = String((error as { message?: string })?.message ?? fallbackMessage)
58	  const schemaIssue = isSchemaCacheError(error)
59	
60	  Sentry.captureException(error, {
61	    tags: {
62	      module: 'cheltuieli',
63	      table: 'cheltuieli_diverse',
64	    },
65	    extra: {
66	      originalMessage: message,
67	      schemaIssue,
68	    },
69	  })
70	
71	  if (schemaIssue) {
72	    hapticError()
73	    toast.error('Schema DB nu e sincronizata. Reincarca aplicatia sau ruleaza reload schema in Supabase.')
74	    return
75	  }
76	
77	  hapticError()
78	  toast.error(message || fallbackMessage)
79	}
