    1: 'use client';
    2: 
    3: import { useState } from 'react';
    4: import { useRouter } from 'next/navigation';
    5: import { Loader2 } from 'lucide-react';
    6: import { getSupabase } from '@/lib/supabase/client';
    7: import { Button } from '@/components/ui/button';
    8: import { Input } from '@/components/ui/input';
    9: import { Label } from '@/components/ui/label';
   10: import { toast } from '@/lib/ui/toast';
   11: 
   12: export default function ResetPasswordPage() {
   13:   const router = useRouter();
   14:   const supabase = getSupabase();
   15:   const [password, setPassword] = useState('');
   16:   const [confirm, setConfirm] = useState('');
   17:   const [loading, setLoading] = useState(false);
   18: 
   19:   const handleReset = async (e: React.FormEvent) => {
   20:     e.preventDefault();
   21: 
   22:     if (password !== confirm) {
   23:       toast.error('Parolele nu coincid!');
   24:       return;
   25:     }
   26: 
   27:     if (password.length < 6) {
   28:       toast.error('Parola trebuie să aibă minim 6 caractere!');
   29:       return;
   30:     }
   31: 
   32:     setLoading(true);
   33: 
   34:     const { error } = await supabase.auth.updateUser({ password });
   35: 
   36:     if (error) {
   37:       toast.error('Eroare la schimbarea parolei. Încearcă din nou.');
   38:     } else {
   39:       toast.success('Parola schimbată cu succes!');
   40:       router.push('/dashboard');
   41:     }
   42: 
   43:     setLoading(false);
   44:   };
   45: 
   46:   return (
   47:     <div className="min-h-screen flex items-center justify-center bg-gray-50">
   48:       <div className="bg-white p-8 rounded-2xl shadow-md w-full max-w-md">
   49:         <div className="text-center mb-8">
   50:           <h1 className="text-3xl font-bold text-[#312E3F]">🫐 Zmeurel OS</h1>
   51:           <p className="text-gray-500 mt-2">Setează o parolă nouă</p>
   52:         </div>
   53: 
   54:         <form onSubmit={handleReset} className="space-y-4">
   55:           <div>
   56:             <Label htmlFor="password">Parolă nouă</Label>
   57:             <Input
   58:               id="password"
   59:               type="password"
   60:               placeholder="Minim 6 caractere"
   61:               value={password}
   62:               onChange={(e) => setPassword(e.target.value)}
   63:               required
   64:             />
   65:           </div>
   66: 
   67:           <div>
   68:             <Label htmlFor="confirm">Confirmă parola</Label>
   69:             <Input
   70:               id="confirm"
   71:               type="password"
   72:               placeholder="Repetă parola"
   73:               value={confirm}
   74:               onChange={(e) => setConfirm(e.target.value)}
   75:               required
   76:             />
   77:           </div>
   78: 
   79:           <Button
   80:             type="submit"
   81:             className="w-full bg-[#F16B6B] hover:bg-[#e05555] text-white"
   82:             disabled={loading}
   83:           >
   84:             {loading ? (
   85:               <>
   86:                 <Loader2 className="h-4 w-4 animate-spin" />
   87:                 Se salvează...
   88:               </>
   89:             ) : (
   90:               'Salvează parola'
   91:             )}
   92:           </Button>
   93:         </form>
   94:       </div>
   95:     </div>
   96:   );
   97: }
