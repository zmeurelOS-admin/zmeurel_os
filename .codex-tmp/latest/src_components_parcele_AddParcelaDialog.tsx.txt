    1: 'use client';
    2: 
    3: import { useState } from 'react';
    4: import { useMutation, useQueryClient } from '@tanstack/react-query';
    5: import { useForm } from 'react-hook-form';
    6: import { zodResolver } from '@hookform/resolvers/zod';
    7: import { z } from 'zod';
    8: import { toast } from '@/lib/ui/toast';
    9: import { Plus } from 'lucide-react';
   10: 
   11: import { AppDialog } from '@/components/app/AppDialog';
   12: import { Button } from '@/components/ui/button';
   13: import { DialogFormActions } from '@/components/ui/dialog-form-actions';
   14: import { createParcela } from '@/lib/supabase/queries/parcele';
   15: import { hapticError, hapticSuccess } from '@/lib/utils/haptic';
   16: import { ParcelaForm, type ParcelaFormData } from './ParcelaForm';
   17: 
   18: const parcelaSchema = z.object({
   19:   nume_parcela: z.string().min(1, 'Numele parcelei este obligatoriu'),
   20:   suprafata_m2: z.string().min(1, 'Suprafata este obligatorie'),
   21:   soi_plantat: z.string().optional(),
   22:   an_plantare: z.string().min(1, 'Anul plantarii este obligatoriu'),
   23:   nr_plante: z.string().optional(),
   24:   status: z.string().default('Activ'),
   25:   observatii: z.string().optional(),
   26: });
   27: 
   28: interface AddParcelaDialogProps {
   29:   soiuriDisponibile: string[];
   30:   onSuccess: () => void;
   31: }
   32: 
   33: export function AddParcelaDialog({ soiuriDisponibile, onSuccess }: AddParcelaDialogProps) {
   34:   const [open, setOpen] = useState(false);
   35:   const queryClient = useQueryClient();
   36: 
   37:   const form = useForm<ParcelaFormData>({
   38:     resolver: zodResolver(parcelaSchema),
   39:     defaultValues: {
   40:       nume_parcela: '',
   41:       suprafata_m2: '',
   42:       soi_plantat: '',
   43:       an_plantare: String(new Date().getFullYear()),
   44:       nr_plante: '',
   45:       status: 'Activ',
   46:       observatii: '',
   47:     },
   48:   });
   49: 
   50:   const createMutation = useMutation({
   51:     mutationFn: async (data: ParcelaFormData) => {
   52:       const idParcela = `PAR-${Date.now().toString().slice(-6)}`;
   53:       return createParcela({
   54:         id_parcela: idParcela,
   55:         nume_parcela: data.nume_parcela,
   56:         suprafata_m2: Number(data.suprafata_m2),
   57:         soi_plantat: data.soi_plantat || undefined,
   58:         an_plantare: Number(data.an_plantare),
   59:         nr_plante: data.nr_plante ? Number(data.nr_plante) : undefined,
   60:         status: data.status,
   61:         observatii: data.observatii || undefined,
   62:       });
   63:     },
   64:     onSuccess: () => {
   65:       queryClient.invalidateQueries({ queryKey: ['parcele'] });
   66:       hapticSuccess();
   67:       toast.success('Parcela a fost adaugata cu succes');
   68:       form.reset();
   69:       setOpen(false);
   70:       onSuccess();
   71:     },
   72:     onError: (error: Error) => {
   73:       hapticError();
   74:       toast.error(`Eroare: ${error.message}`);
   75:     },
   76:   });
   77: 
   78:   const onSubmit = (data: ParcelaFormData) => {
   79:     createMutation.mutate(data);
   80:   };
   81: 
   82:   return (
   83:     <>
   84:       <Button size="lg" className="h-14 w-full rounded-2xl shadow-sm" onClick={() => setOpen(true)}>
   85:         <Plus className="mr-2 h-5 w-5" />
   86:         Adauga Parcela
   87:       </Button>
   88: 
   89:       <AppDialog
   90:         open={open}
   91:         onOpenChange={setOpen}
   92:         title="Adauga Parcela Noua"
   93:         description="Completeaza detaliile parcelei."
   94:         footer={
   95:           <DialogFormActions
   96:             onCancel={() => setOpen(false)}
   97:             onSave={form.handleSubmit(onSubmit)}
   98:             saving={createMutation.isPending}
   99:             cancelLabel="Anuleaza"
  100:             saveLabel="Salveaza parcela"
  101:           />
  102:         }
  103:       >
  104:         <form id="add-parcela-form" onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
  105:           <ParcelaForm form={form} soiuriDisponibile={soiuriDisponibile} />
  106:         </form>
  107:       </AppDialog>
  108:     </>
  109:   );
  110: }
