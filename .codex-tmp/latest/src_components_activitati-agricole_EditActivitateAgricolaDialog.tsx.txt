    1: 'use client'
    2: 
    3: import { useEffect } from 'react'
    4: import { zodResolver } from '@hookform/resolvers/zod'
    5: import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query'
    6: import { useForm } from 'react-hook-form'
    7: import { toast } from '@/lib/ui/toast'
    8: import * as z from 'zod'
    9: 
   10: import { AppDialog } from '@/components/app/AppDialog'
   11: import { DialogFormActions } from '@/components/ui/dialog-form-actions'
   12: import { Input } from '@/components/ui/input'
   13: import { Label } from '@/components/ui/label'
   14: import { Textarea } from '@/components/ui/textarea'
   15: import { track } from '@/lib/analytics/track'
   16: import { getParcele } from '@/lib/supabase/queries/parcele'
   17: import {
   18:   type ActivitateAgricola,
   19:   updateActivitateAgricola,
   20: } from '@/lib/supabase/queries/activitati-agricole'
   21: import { hapticError, hapticSuccess } from '@/lib/utils/haptic'
   22: 
   23: const schema = z.object({
   24:   data_aplicare: z.string().min(1, 'Data este obligatorie'),
   25:   parcela_id: z.string().optional(),
   26:   tip_activitate: z.string().min(1, 'Tipul activitatii este obligatoriu'),
   27:   produs_utilizat: z.string().optional(),
   28:   doza: z.string().optional(),
   29:   timp_pauza_zile: z
   30:     .string()
   31:     .trim()
   32:     .optional()
   33:     .refine((value) => !value || (Number.isFinite(Number(value)) && Number(value) >= 0), {
   34:       message: 'Timpul de pauza trebuie sa fie un numar valid',
   35:     }),
   36:   observatii: z.string().optional(),
   37: })
   38: 
   39: type FormValues = z.infer<typeof schema>
   40: 
   41: interface EditActivitateAgricolaDialogProps {
   42:   activitate: ActivitateAgricola | null
   43:   open: boolean
   44:   onOpenChange: (open: boolean) => void
   45: }
   46: 
   47: const defaults = (): FormValues => ({
   48:   data_aplicare: new Date().toISOString().split('T')[0],
   49:   parcela_id: '',
   50:   tip_activitate: '',
   51:   produs_utilizat: '',
   52:   doza: '',
   53:   timp_pauza_zile: '0',
   54:   observatii: '',
   55: })
   56: 
   57: export function EditActivitateAgricolaDialog({
   58:   activitate,
   59:   open,
   60:   onOpenChange,
   61: }: EditActivitateAgricolaDialogProps) {
   62:   const queryClient = useQueryClient()
   63: 
   64:   const form = useForm<FormValues>({
   65:     resolver: zodResolver(schema),
   66:     defaultValues: defaults(),
   67:   })
   68: 
   69:   useEffect(() => {
   70:     if (!open || !activitate) return
   71: 
   72:     form.reset({
   73:       data_aplicare: activitate.data_aplicare
   74:         ? new Date(activitate.data_aplicare).toISOString().split('T')[0]
   75:         : new Date().toISOString().split('T')[0],
   76:       parcela_id: activitate.parcela_id ?? '',
   77:       tip_activitate: activitate.tip_activitate ?? '',
   78:       produs_utilizat: activitate.produs_utilizat ?? '',
   79:       doza: activitate.doza ?? '',
   80:       timp_pauza_zile:
   81:         typeof activitate.timp_pauza_zile === 'number' ? String(activitate.timp_pauza_zile) : '0',
   82:       observatii: activitate.observatii ?? '',
   83:     })
   84:   }, [open, activitate, form])
   85: 
   86:   const { data: parcele = [] } = useQuery({
   87:     queryKey: ['parcele'],
   88:     queryFn: getParcele,
   89:   })
   90: 
   91:   const mutation = useMutation({
   92:     mutationFn: ({ id, values }: { id: string; values: FormValues }) =>
   93:       updateActivitateAgricola(id, {
   94:         data_aplicare: values.data_aplicare,
   95:         parcela_id: values.parcela_id || undefined,
   96:         tip_activitate: values.tip_activitate,
   97:         produs_utilizat: values.produs_utilizat?.trim() || undefined,
   98:         doza: values.doza?.trim() || undefined,
   99:         timp_pauza_zile: Number(values.timp_pauza_zile || 0),
  100:         observatii: values.observatii?.trim() || undefined,
  101:       }),
  102:     onSuccess: (_, variables) => {
  103:       queryClient.invalidateQueries({ queryKey: ['activitati'] })
  104:       track('activitate_edit', { id: variables.id })
  105:       hapticSuccess()
  106:       toast.success('Activitate actualizata')
  107:       onOpenChange(false)
  108:     },
  109:     onError: (error: any) => {
  110:       hapticError()
  111:       toast.error(error?.message || 'Eroare la actualizare')
  112:     },
  113:   })
  114: 
  115:   if (!activitate) return null
  116: 
  117:   const submit = (values: FormValues) => {
  118:     if (mutation.isPending) return
  119:     mutation.mutate({ id: activitate.id, values })
  120:   }
  121: 
  122:   return (
  123:     <AppDialog
  124:       open={open}
  125:       onOpenChange={onOpenChange}
  126:       title="Editeaza activitate agricola"
  127:       footer={
  128:         <DialogFormActions
  129:           onCancel={() => onOpenChange(false)}
  130:           onSave={form.handleSubmit(submit)}
  131:           saving={mutation.isPending}
  132:           cancelLabel="Anuleaza"
  133:           saveLabel="Salveaza"
  134:         />
  135:       }
  136:     >
  137:       <form className="space-y-4" onSubmit={form.handleSubmit(submit)}>
  138:         <div className="space-y-2">
  139:           <Label htmlFor="act_edit_data">Data aplicare</Label>
  140:           <Input id="act_edit_data" type="date" className="agri-control h-12" {...form.register('data_aplicare')} />
  141:           {form.formState.errors.data_aplicare ? <p className="text-xs text-red-600">{form.formState.errors.data_aplicare.message}</p> : null}
  142:         </div>
  143: 
  144:         <div className="space-y-2">
  145:           <Label htmlFor="act_edit_parcela">Parcela</Label>
  146:           <select id="act_edit_parcela" className="agri-control h-12 w-full px-3 text-base" {...form.register('parcela_id')}>
  147:             <option value="">Selecteaza parcela</option>
  148:             {parcele.map((parcela: any) => (
  149:               <option key={parcela.id} value={parcela.id}>
  150:                 {parcela.nume_parcela || 'Parcela'}
  151:               </option>
  152:             ))}
  153:           </select>
  154:         </div>
  155: 
  156:         <div className="space-y-2">
  157:           <Label htmlFor="act_edit_tip">Tip activitate</Label>
  158:           <select id="act_edit_tip" className="agri-control h-12 w-full px-3 text-base" {...form.register('tip_activitate')}>
  159:             <option value="">Tip operatiune</option>
  160:             <option value="fertilizare_foliara">Fertilizare foliara</option>
  161:             <option value="fertirigare">Fertirigare</option>
  162:             <option value="fertilizare_baza">Fertilizare de baza</option>
  163:             <option value="fungicide_pesticide">Fungicide/Pesticide</option>
  164:             <option value="irigatie">Irigatie</option>
  165:             <option value="altele">Altele</option>
  166:           </select>
  167:           {form.formState.errors.tip_activitate ? <p className="text-xs text-red-600">{form.formState.errors.tip_activitate.message}</p> : null}
  168:         </div>
  169: 
  170:         <div className="space-y-2">
  171:           <Label htmlFor="act_edit_produs">Produs</Label>
  172:           <Input id="act_edit_produs" className="agri-control h-12" {...form.register('produs_utilizat')} />
  173:         </div>
  174: 
  175:         <div className="space-y-2">
  176:           <Label htmlFor="act_edit_doza">Cantitate / doza</Label>
  177:           <Input id="act_edit_doza" className="agri-control h-12" {...form.register('doza')} />
  178:         </div>
  179: 
  180:         <div className="space-y-2">
  181:           <Label htmlFor="act_edit_pauza">Timp pauza (zile)</Label>
  182:           <Input id="act_edit_pauza" type="number" min="0" className="agri-control h-12" {...form.register('timp_pauza_zile')} />
  183:           {form.formState.errors.timp_pauza_zile ? <p className="text-xs text-red-600">{form.formState.errors.timp_pauza_zile.message}</p> : null}
  184:         </div>
  185: 
  186:         <div className="space-y-2">
  187:           <Label htmlFor="act_edit_obs">Observatii</Label>
  188:           <Textarea id="act_edit_obs" rows={4} className="agri-control w-full px-3 py-2 text-base" {...form.register('observatii')} />
  189:         </div>
  190:       </form>
  191:     </AppDialog>
  192:   )
  193: }
