    1: 'use client'
    2: 
    3: import { FormEvent, useMemo, useState } from 'react'
    4: import { Loader2 } from 'lucide-react'
    5: import { useRouter } from 'next/navigation'
    6: import { toast } from '@/lib/ui/toast'
    7: 
    8: import { Button } from '@/components/ui/button'
    9: import { Input } from '@/components/ui/input'
   10: import { Label } from '@/components/ui/label'
   11: import { getSupabase } from '@/lib/supabase/client'
   12: 
   13: export default function UpdatePasswordPage() {
   14:   const router = useRouter()
   15:   const supabase = useMemo(() => getSupabase(), [])
   16:   const [password, setPassword] = useState('')
   17:   const [confirmPassword, setConfirmPassword] = useState('')
   18:   const [loading, setLoading] = useState(false)
   19: 
   20:   const handleSubmit = async (event: FormEvent<HTMLFormElement>) => {
   21:     event.preventDefault()
   22: 
   23:     if (password.length < 8) {
   24:       toast.error('Parola trebuie să aibă minim 8 caractere.')
   25:       return
   26:     }
   27: 
   28:     if (password !== confirmPassword) {
   29:       toast.error('Parolele nu coincid.')
   30:       return
   31:     }
   32: 
   33:     setLoading(true)
   34:     const { error } = await supabase.auth.updateUser({ password })
   35:     setLoading(false)
   36: 
   37:     if (error) {
   38:       toast.error(error.message)
   39:       return
   40:     }
   41: 
   42:     toast.success('Parola a fost actualizată.')
   43:     router.push('/dashboard')
   44:   }
   45: 
   46:   return (
   47:     <div className="flex min-h-screen items-center justify-center bg-gray-50 p-4">
   48:       <div className="w-full max-w-md rounded-2xl bg-white p-6 shadow-md sm:p-8">
   49:         <h1 className="text-2xl font-bold text-[#312E3F]">Actualizare parolă</h1>
   50:         <p className="mt-2 text-sm text-gray-500">Setează o parolă nouă pentru cont.</p>
   51: 
   52:         <form onSubmit={handleSubmit} className="mt-5 space-y-4">
   53:           <div className="space-y-2">
   54:             <Label htmlFor="new-password">Parolă nouă</Label>
   55:             <Input
   56:               id="new-password"
   57:               type="password"
   58:               autoComplete="new-password"
   59:               placeholder="Minim 8 caractere"
   60:               value={password}
   61:               onChange={(event) => setPassword(event.target.value)}
   62:               required
   63:             />
   64:           </div>
   65: 
   66:           <div className="space-y-2">
   67:             <Label htmlFor="confirm-password">Confirmă parola</Label>
   68:             <Input
   69:               id="confirm-password"
   70:               type="password"
   71:               autoComplete="new-password"
   72:               placeholder="Repetă parola"
   73:               value={confirmPassword}
   74:               onChange={(event) => setConfirmPassword(event.target.value)}
   75:               required
   76:             />
   77:           </div>
   78: 
   79:           <Button type="submit" className="w-full bg-[#F16B6B] text-white hover:bg-[#e05555]" disabled={loading}>
   80:             {loading ? (
   81:               <>
   82:                 <Loader2 className="h-4 w-4 animate-spin" />
   83:                 Se actualizeaza...
   84:               </>
   85:             ) : (
   86:               'Salveaza parola'
   87:             )}
   88:           </Button>
   89:         </form>
   90:       </div>
   91:     </div>
   92:   )
   93: }
