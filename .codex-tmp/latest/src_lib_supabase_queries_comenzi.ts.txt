  134:   const outflowTypes = new Set(['vanzare', 'consum', 'oferit_gratuit', 'pierdere'])

  138: async function getStockBuckets(): Promise<StockBucket[]> {
  139:   const supabase = getSupabase()
  140:   const { data, error } = await (supabase as any)
  141:     .from('miscari_stoc')
  142:     .select('locatie_id,produs,calitate,depozit,tip_miscare,cantitate_kg')
  143: 
  144:   if (error) throw toReadableError(error, 'Nu am putut incarca stocul disponibil.')
  145: 
  146:   const grouped = new Map<string, StockBucket>()
  147:   for (const row of data ?? []) {
  148:     const key = `${row.locatie_id}|${row.produs}|${row.calitate}|${row.depozit}`
  149:     const existing = grouped.get(key) ?? {
  150:       locatie_id: row.locatie_id,
  151:       produs: row.produs,
  152:       calitate: row.calitate,
  153:       depozit: row.depozit,
  154:       availableKg: 0,
  155:     }
  156:     const qty = signedQuantity(String(row.tip_miscare), Number(row.cantitate_kg ?? 0))
  157:     existing.availableKg = round2(existing.availableKg + qty)
  158:     grouped.set(key, existing)
  159:   }
  160: 
  161:   return Array.from(grouped.values())
  162:     .filter((bucket) => bucket.availableKg > 0)
  163:     .sort((a, b) => b.availableKg - a.availableKg)
  164: }

  166: async function applyStockOutflowForVanzare(vanzareId: string, cantitateKg: number, tenantId: string): Promise<number> {
  167:   const supabase = getSupabase()
  168:   let remaining = round2(cantitateKg)
  169:   if (remaining <= 0) return 0
  170: 
  171:   const buckets = await getStockBuckets()
  172:   let deducted = 0
  173: 
  174:   for (const bucket of buckets) {
  175:     if (remaining <= 0) break
  176:     const take = round2(Math.min(bucket.availableKg, remaining))
  177:     if (take <= 0) continue
  178: 
  179:     const payload = {
  180:       tenant_id: tenantId,
  181:       locatie_id: bucket.locatie_id,
  182:       produs: bucket.produs,
  183:       calitate: bucket.calitate,
  184:       depozit: bucket.depozit,
  185:       tip_miscare: 'vanzare',
  186:       cantitate_kg: take,
  187:       referinta_id: vanzareId,
  188:       data: new Date().toISOString().split('T')[0],
  189:       observatii: 'Consum stoc prin livrare comanda',
  190:     }
  191: 
  192:     const { error } = await (supabase as any).from('miscari_stoc').insert(payload)
  193:     if (error) {
  194:       throw toReadableError(error, 'Nu am putut actualiza stocul pentru livrare.')
  195:     }
  196: 
  197:     deducted = round2(deducted + take)
  198:     remaining = round2(remaining - take)
  199:   }
  200: 
  201:   return deducted
  202: }

  377: export async function deleteComanda(id: string): Promise<void> {
  378:   const supabase = getSupabase()
  379:   
  380:   // First, fetch the comanda to check if it has a linked vanzare
  381:   const { data: comanda, error: fetchError } = await (supabase as any)
  382:     .from('comenzi')
  383:     .select('linked_vanzare_id, status')
  384:     .eq('id', id)
  385:     .single()
  386:   
  387:   if (fetchError) throw toReadableError(fetchError, 'Nu am putut incarca comanda pentru stergere.')
  388:   
  389:   // If comanda has linked_vanzare_id or status='livrata', delete the linked vanzare first
  390:   // This creates a cascade: comanda delete → vanzare delete → stock movements delete
  391:   if (comanda.linked_vanzare_id || comanda.status === 'livrata') {
  392:     if (comanda.linked_vanzare_id) {
  393:       // deleteVanzare already handles deleting stock movements (miscari_stoc)
  394:       await deleteVanzare(comanda.linked_vanzare_id)
  395:     }
  396:   }
  397:   
  398:   // Finally, delete the comanda
  399:   const { error } = await (supabase as any).from('comenzi').delete().eq('id', id)
  400:   if (error) throw toReadableError(error, 'Nu am putut sterge comanda.')
  401: }

  403: export async function deliverComanda(input: DeliverComandaInput): Promise<{

  426:   if (current.status === 'anulata') {

  447:

  449:     .from('comenzi')
  450:     .update({
  451:       status: 'livrata',

  472:       parent_comanda_id,
  473:       created_at,
  474:       updated_at,
  475:       clienti (
  476:         nume_client
  477:       )
  478:     `)

  480: 
  481:   if (deliveredError) throw toReadableError(deliveredError, 'Nu am putut actualiza comanda ca livrata.')
  482: 
  483:   let remainingOrder: Comanda | null = null

  487:     fallbackDate.setDate(fallbackDate.getDate() + 1)

  489:     const today = new Date().toISOString().split('T')[0]
  490: 
  491:     const { data: remainingData, error: remainingError } = await (supabase as any)

  501:         cantitate_kg: remainingQty,
  502:         pret_per_kg: Number(current.pret_per_kg ?? 0),
  503:         total: round2(remainingQty * Number(current.pret_per_kg ?? 0)),
  504:         status: remainingDate > today ? 'programata' : 'confirmata',
  505:         observatii: [current.observatii, `Rest din comanda ${current.id}`].filter(Boolean).join(' | '),
  506:         parent_comanda_id: current.id,
  507:       })
  508:       .select(`

  531:
