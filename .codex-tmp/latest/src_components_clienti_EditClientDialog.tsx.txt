    1: 'use client'
    2: 
    3: import { useEffect, useState } from 'react'
    4: import { zodResolver } from '@hookform/resolvers/zod'
    5: import { Loader2 } from 'lucide-react'
    6: import { useForm } from 'react-hook-form'
    7: import * as z from 'zod'
    8: 
    9: import { AppDialog } from '@/components/app/AppDialog'
   10: import { Button } from '@/components/ui/button'
   11: import { Input } from '@/components/ui/input'
   12: import { Label } from '@/components/ui/label'
   13: import { Textarea } from '@/components/ui/textarea'
   14: import type { Client } from '@/lib/supabase/queries/clienti'
   15: 
   16: const schema = z.object({
   17:   nume_client: z.string().trim().min(2, 'Numele trebuie sa aiba minim 2 caractere'),
   18:   telefon: z.string().optional(),
   19:   email: z.string().email('Email invalid').or(z.literal('')).optional(),
   20:   adresa: z.string().optional(),
   21:   pret_negociat_lei_kg: z.string().optional(),
   22:   observatii: z.string().optional(),
   23: })
   24: 
   25: type ClientFormData = z.infer<typeof schema>
   26: 
   27: interface EditClientDialogProps {
   28:   client: Client | null
   29:   open: boolean
   30:   onOpenChange: (open: boolean) => void
   31:   onSubmit: (id: string, data: ClientFormData) => Promise<void>
   32: }
   33: 
   34: const defaults = (): ClientFormData => ({
   35:   nume_client: '',
   36:   telefon: '',
   37:   email: '',
   38:   adresa: '',
   39:   pret_negociat_lei_kg: '',
   40:   observatii: '',
   41: })
   42: 
   43: export function EditClientDialog({ client, open, onOpenChange, onSubmit }: EditClientDialogProps) {
   44:   const [isSubmitting, setIsSubmitting] = useState(false)
   45: 
   46:   const form = useForm<ClientFormData>({
   47:     resolver: zodResolver(schema),
   48:     defaultValues: defaults(),
   49:   })
   50: 
   51:   useEffect(() => {
   52:     if (!open || !client) return
   53:     form.reset({
   54:       nume_client: client.nume_client || '',
   55:       telefon: client.telefon || '',
   56:       email: client.email || '',
   57:       adresa: client.adresa || '',
   58:       pret_negociat_lei_kg: client.pret_negociat_lei_kg ? String(client.pret_negociat_lei_kg) : '',
   59:       observatii: client.observatii || '',
   60:     })
   61:   }, [open, client, form])
   62: 
   63:   const handleSubmit = async (data: ClientFormData) => {
   64:     if (!client) return
   65:     setIsSubmitting(true)
   66:     try {
   67:       await onSubmit(client.id, data)
   68:       onOpenChange(false)
   69:     } finally {
   70:       setIsSubmitting(false)
   71:     }
   72:   }
   73: 
   74:   if (!client) return null
   75: 
   76:   return (
   77:     <AppDialog
   78:       open={open}
   79:       onOpenChange={onOpenChange}
   80:       title="Editeaza client"
   81:       footer={
   82:         <div className="grid grid-cols-2 gap-3">
   83:           <Button type="button" variant="outline" className="agri-cta" onClick={() => onOpenChange(false)} disabled={isSubmitting}>
   84:             Anuleaza
   85:           </Button>
   86:           <Button
   87:             type="button"
   88:             className="agri-cta bg-[var(--agri-primary)] text-white hover:bg-emerald-700"
   89:             onClick={form.handleSubmit(handleSubmit)}
   90:             disabled={isSubmitting}
   91:           >
   92:             {isSubmitting ? (
   93:               <>
   94:                 <Loader2 className="h-4 w-4 animate-spin" />
   95:                 Se salveaza...
   96:               </>
   97:             ) : (
   98:               'Salveaza'
   99:             )}
  100:           </Button>
  101:         </div>
  102:       }
  103:     >
  104:       <form className="space-y-4" onSubmit={form.handleSubmit(handleSubmit)}>
  105:         <div className="space-y-2">
  106:           <Label htmlFor="edit_client_nume">Nume client</Label>
  107:           <Input id="edit_client_nume" className="agri-control h-12" {...form.register('nume_client')} />
  108:           {form.formState.errors.nume_client ? <p className="text-xs text-red-600">{form.formState.errors.nume_client.message}</p> : null}
  109:         </div>
  110: 
  111:         <div className="space-y-2">
  112:           <Label htmlFor="edit_client_telefon">Telefon</Label>
  113:           <Input id="edit_client_telefon" type="tel" className="agri-control h-12" {...form.register('telefon')} />
  114:         </div>
  115: 
  116:         <div className="space-y-2">
  117:           <Label htmlFor="edit_client_email">Email</Label>
  118:           <Input id="edit_client_email" type="email" className="agri-control h-12" {...form.register('email')} />
  119:           {form.formState.errors.email ? <p className="text-xs text-red-600">{form.formState.errors.email.message}</p> : null}
  120:         </div>
  121: 
  122:         <div className="space-y-2">
  123:           <Label htmlFor="edit_client_adresa">Adresa</Label>
  124:           <Textarea id="edit_client_adresa" rows={3} className="agri-control w-full px-3 py-2 text-base" {...form.register('adresa')} />
  125:         </div>
  126: 
  127:         <div className="space-y-2">
  128:           <Label htmlFor="edit_client_pret">Pret negociat (lei/kg)</Label>
  129:           <Input
  130:             id="edit_client_pret"
  131:             type="number"
  132:             inputMode="decimal"
  133:             step="0.01"
  134:             min="0"
  135:             className="agri-control h-12"
  136:             {...form.register('pret_negociat_lei_kg')}
  137:           />
  138:         </div>
  139: 
  140:         <div className="space-y-2">
  141:           <Label htmlFor="edit_client_obs">Observatii</Label>
  142:           <Textarea id="edit_client_obs" rows={4} className="agri-control w-full px-3 py-2 text-base" {...form.register('observatii')} />
  143:         </div>
  144:       </form>
  145:     </AppDialog>
  146:   )
  147: }
