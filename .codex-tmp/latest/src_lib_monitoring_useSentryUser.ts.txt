   13:           supabase.auth.getUser(),

   17:         if (!data?.user) return

   19:         Sentry.setUser({

   21:           email: data.user.email ?? undefined,
   22:         })
   23:       } catch {
   24:         // monitoring must never affect app runtime
   25:       }
   26:     })()
   27:   }, [])
   28: }
