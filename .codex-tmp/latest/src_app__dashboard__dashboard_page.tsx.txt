   23: function formatCurrency(value: number) {
   24:   return new Intl.NumberFormat('ro-RO', {
   25:     style: 'currency',
   26:     currency: 'RON',
   27:     maximumFractionDigits: 0,
   28:   }).format(value)
   29: }
   30: 
   31: function toDateOnly(value: string | null | undefined): string {
   32:   return (value ?? '').slice(0, 10)
   33: }
   34: 
   35: function getTrendDirection(current: number, previous: number): 'up' | 'down' | 'neutral' {
   36:   const delta = current - previous
   37:   if (Math.abs(delta) <= 0.5) return 'neutral'
   38:   if (delta > 0) return 'up'
   39:   if (delta < 0) return 'down'
   40:   return 'neutral'
   41: }
