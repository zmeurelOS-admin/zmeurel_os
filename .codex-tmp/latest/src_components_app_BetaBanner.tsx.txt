    1: 'use client'
    2: 
    3: import { useMemo, useState } from 'react'
    4: import { usePathname } from 'next/navigation'
    5: import { Loader2 } from 'lucide-react'
    6: 
    7: import { AppDialog } from '@/components/app/AppDialog'
    8: import { useDashboardAuth } from '@/components/app/DashboardAuthContext'
    9: import { Button } from '@/components/ui/button'
   10: import { Textarea } from '@/components/ui/textarea'
   11: import { track } from '@/lib/analytics/track'
   12: import { getSupabase } from '@/lib/supabase/client'
   13: import { toast } from '@/lib/ui/toast'
   14: 
   15: const RATING_OPTIONS = [
   16:   { value: 1, emoji: ':(' },
   17:   { value: 2, emoji: ':/' },
   18:   { value: 3, emoji: ':|' },
   19:   { value: 4, emoji: ':)' },
   20:   { value: 5, emoji: '<3' },
   21: ] as const
   22: 
   23: export function BetaBanner() {
   24:   const { userId } = useDashboardAuth()
   25:   const pathname = usePathname()
   26:   const [feedbackOpen, setFeedbackOpen] = useState(false)
   27:   const [message, setMessage] = useState('')
   28:   const [rating, setRating] = useState<number>(4)
   29:   const [submitting, setSubmitting] = useState(false)
   30: 
   31:   const canSubmit = useMemo(() => message.trim().length > 0 && rating >= 1 && rating <= 5, [message, rating])
   32: 
   33:   const submitFeedback = async () => {
   34:     if (!userId || !canSubmit || submitting) return
   35:     setSubmitting(true)
   36: 
   37:     try {
   38:       const supabase = getSupabase()
   39:       const { data: tenant, error: tenantError } = await supabase
   40:         .from('tenants')
   41:         .select('id')
   42:         .eq('owner_user_id', userId)
   43:         .maybeSingle()
   44: 
   45:       if (tenantError) throw tenantError
   46:       if (!tenant?.id) throw new Error('Tenant indisponibil pentru feedback.')
   47: 
   48:       const { error: insertError } = await supabase
   49:         .from('feedback' as never)
   50:         .insert({
   51:           tenant_id: tenant.id,
   52:           user_id: userId,
   53:           rating,
   54:           message: message.trim(),
   55:           page_url: pathname || '/',
   56:         } as never)
   57: 
   58:       if (insertError) throw insertError
   59: 
   60:       track('feedback_sent', { rating, page: pathname || '/' })
   61: 
   62:       toast.success('Multumim pentru feedback!')
   63:       setMessage('')
   64:       setRating(4)
   65:       setFeedbackOpen(false)
   66:     } catch (error) {
   67:       const resolved = error instanceof Error ? error.message : 'Nu am putut trimite feedback-ul.'
   68:       toast.error(resolved)
   69:     } finally {
   70:       setSubmitting(false)
   71:     }
   72:   }
   73: 
   74:   return (
   75:     <>
   76:       <div className="sticky top-0 z-50 border-b border-amber-200 bg-amber-100/95 text-amber-900 backdrop-blur">
   77:         <div className="mx-auto flex min-h-9 w-full max-w-7xl items-center justify-between gap-2 px-3 py-1.5 sm:px-4">
   78:           <p className="text-xs font-medium sm:text-sm">Zmeurel OS este in BETA - Trimite feedback</p>
   79:           <Button
   80:             type="button"
   81:             variant="outline"
   82:             className="h-7 border-amber-300 bg-amber-50 px-2.5 text-xs text-amber-900 hover:bg-amber-200"
   83:             onClick={() => {
   84:               track('feedback_banner_click', { page: pathname || '/' })
   85:               setFeedbackOpen(true)
   86:             }}
   87:           >
   88:             Trimite feedback
   89:           </Button>
   90:         </div>
   91:       </div>
   92: 
   93:       <AppDialog
   94:         open={feedbackOpen}
   95:         onOpenChange={setFeedbackOpen}
   96:         title="Trimite feedback"
   97:         footer={
   98:           <>
   99:             <Button type="button" variant="outline" onClick={() => setFeedbackOpen(false)}>
  100:               Anuleaza
  101:             </Button>
  102:             <Button type="button" onClick={submitFeedback} disabled={!canSubmit || submitting}>
  103:               {submitting ? (
  104:                 <>
  105:                   <Loader2 className="h-4 w-4 animate-spin" />
  106:                   Se trimite...
  107:                 </>
  108:               ) : (
  109:                 'Trimite'
  110:               )}
  111:             </Button>
  112:           </>
  113:         }
  114:       >
  115:         <div className="space-y-4">
  116:           <Textarea
  117:             placeholder="Ce functioneaza bine? Ce ai imbunatati?"
  118:             value={message}
  119:             onChange={(event) => setMessage(event.target.value)}
  120:             className="min-h-28"
  121:           />
  122: 
  123:           <div className="space-y-2">
  124:             <p className="text-sm font-medium text-[var(--agri-text)]">Cum ti se pare experienta?</p>
  125:             <div className="flex items-center gap-2">
  126:               {RATING_OPTIONS.map((option) => {
  127:                 const selected = rating === option.value
  128:                 return (
  129:                   <button
  130:                     key={option.value}
  131:                     type="button"
  132:                     aria-label={`Rating ${option.value}`}
  133:                     className={`inline-flex h-10 w-10 items-center justify-center rounded-full border text-xl ${
  134:                       selected
  135:                         ? 'border-amber-600 bg-amber-100'
  136:                         : 'border-[var(--agri-border)] bg-white hover:bg-[var(--agri-surface-muted)]'
  137:                     }`}
  138:                     onClick={() => setRating(option.value)}
  139:                   >
  140:                     {option.emoji}
  141:                   </button>
  142:                 )
  143:               })}
  144:             </div>
  145:           </div>
  146:         </div>
  147:       </AppDialog>
  148:     </>
  149:   )
  150: }
  151:

  153:   // preserved for compatibility with existing imports/tests
  154: }
