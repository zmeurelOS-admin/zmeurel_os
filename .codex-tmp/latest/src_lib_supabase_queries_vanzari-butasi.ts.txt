  116: function extractErrorParts(error: unknown): SupabaseLikeError {
  117:   if (error instanceof Error) {
  118:     const withCause = error as Error & { cause?: unknown }
  119:     const causeParts = withCause.cause ? extractErrorParts(withCause.cause) : {}
  120: 
  121:     return {
  122:       message: error.message || causeParts.message,
  123:       code: causeParts.code,
  124:       details: causeParts.details,
  125:       hint: causeParts.hint,
  126:       statusText: causeParts.statusText,
  127:       error: causeParts.error,
  128:       cause: withCause.cause,
  129:     }
  130:   }
  131: 
  132:   const e = (error ?? {}) as SupabaseLikeError
  133:   if (e.error) {
  134:     const nested = extractErrorParts(e.error)
  135:     return {
  136:       ...nested,
  137:       message: e.message || nested.message,
  138:       code: e.code || nested.code,
  139:       details: e.details || nested.details,
  140:       hint: e.hint || nested.hint,
  141:       statusText: e.statusText || nested.statusText,
  142:       error: e.error,
  143:       cause: e.cause,
  144:     }
  145:   }
  146: 
  147:   if (e.cause) {
  148:     const nested = extractErrorParts(e.cause)
  149:     return {
  150:       ...nested,
  151:       message: e.message || nested.message,
  152:       code: e.code || nested.code,
  153:       details: e.details || nested.details,
  154:       hint: e.hint || nested.hint,
  155:       statusText: e.statusText || nested.statusText,
  156:       error: e.error,
  157:       cause: e.cause,
  158:     }
  159:   }
  160: 
  161:   return e
  162: }

  164: function isSchemaCacheMismatch(error: unknown): boolean {
  165:   const e = extractErrorParts(error)
  166:   const message = (e.message ?? '').toLowerCase()
  167:   return (
  168:     e?.code === 'PGRST204' ||
  169:     e?.code === '42703' ||
  170:     message.includes('schema cache') ||
  171:     message.includes('could not find the')
  172:   )

  175: function toReadableError(error: unknown, fallbackMessage: string): Error & {
  176:   code?: string
  177:   details?: string
  178:   hint?: string
  179:   isSchemaCacheError?: boolean
  180: } {
  181:   const e = extractErrorParts(error)
  182: 
  183:   let serialized = ''
  184:   if (!e.message && !e.details && !e.hint && error && typeof error === 'object') {
  185:     try {
  186:       serialized = JSON.stringify(error)
  187:     } catch {
  188:       serialized = ''
  189:     }
  190:   }
  191: 
  192:   const message =
  193:     e?.message ||
  194:     e?.details ||
  195:     e?.hint ||
  196:     e?.statusText ||
  197:     serialized ||
  198:     (error instanceof Error && error.message) ||
  199:     fallbackMessage
  200: 
  201:   return Object.assign(new Error(message), {
  202:     code: e?.code,
  203:     details: e?.details,
  204:     hint: e?.hint,
  205:     isSchemaCacheError: isSchemaCacheMismatch(error),
  206:   })
  207: }

  221: function normalizeItems(items: VanzareButasiItemInput[]): NormalizedItemsResult {
  222:   if (!Array.isArray(items) || items.length === 0) {
  223:     throw new Error('Comanda trebuie sa contina cel putin un produs')
  224:   }
  225: 
  226:   const normalized = items.map((item, index) => {
  227:     const soi = item.soi?.trim()
  228:     const cantitate = Number(item.cantitate)
  229:     const pretUnitar = Number(item.pret_unitar)
  230: 
  231:     if (!soi) {
  232:       throw new Error(`Soi invalid la linia ${index + 1}`)
  233:     }
  234: 
  235:     if (!Number.isInteger(cantitate) || cantitate <= 0) {
  236:       throw new Error(`Cantitate invalida la linia ${index + 1}`)
  237:     }
  238: 
  239:     if (!Number.isFinite(pretUnitar) || pretUnitar <= 0) {
  240:       throw new Error(`Pret invalid la linia ${index + 1}`)
  241:     }
  242: 
  243:     return {
  244:       soi,
  245:       cantitate,
  246:       pret_unitar: roundTo2(pretUnitar),
  247:       subtotal: roundTo2(cantitate * pretUnitar),
  248:     }
  249:   })
  250: 
  251:   const totalLei = roundTo2(normalized.reduce((sum, item) => sum + item.subtotal, 0))
  252:   const totalCantitate = normalized.reduce((sum, item) => sum + item.cantitate, 0)
  253: 
  254:   return { items: normalized, totalLei, totalCantitate }
  255: }

  354: export async function createVanzareButasi(input: CreateVanzareButasiInput): Promise<VanzareButasi> {
  355:   const supabase = getSupabase()
  356:   const normalized = normalizeItems(input.items)
  357:   const avansSuma = roundTo2(Number(input.avans_suma ?? 0))
  358: 
  359:   if (!Number.isFinite(avansSuma) || avansSuma < 0) {
  360:     throw new Error('Avansul trebuie sa fie un numar pozitiv')
  361:   }
  362: 
  363:   const status = assertStatus(input.status ?? 'noua')
  364:   const firstItem = normalized.items[0]
  365: 
  366:   const { data, error } = await supabase
  367:     .from('vanzari_butasi')
  368:     .insert({
  369:       data: input.data_comanda,
  370:       data_comanda: input.data_comanda,
  371:       data_livrare_estimata: input.data_livrare_estimata || null,
  372:       status,
  373:       client_id: input.client_id || null,
  374:       parcela_sursa_id: input.parcela_sursa_id || null,
  375:       adresa_livrare: input.adresa_livrare?.trim() || null,
  376:       observatii: input.observatii?.trim() || null,
  377:       avans_suma: avansSuma,
  378:       avans_data: input.avans_data || null,
  379:       total_lei: normalized.totalLei,
  380:       soi_butasi: firstItem?.soi ?? null,
  381:       cantitate_butasi: normalized.totalCantitate,
  382:       pret_unitar_lei: roundTo2(normalized.totalLei / normalized.totalCantitate),
  383:     })
  384:     .select('id')
  385:     .single()
  386: 
  387:   if (error) throw toReadableError(error, 'Nu am putut salva comanda de butasi.')
  388: 
  389:   const comandaId = data.id
  390:   const itemsPayload = normalized.items.map((item) => ({
  391:     comanda_id: comandaId,
  392:     soi: item.soi,
  393:     cantitate: item.cantitate,
  394:     pret_unitar: item.pret_unitar,
  395:     subtotal: item.subtotal,
  396:   }))
  397: 
  398:   const { error: itemsError } = await supabase.from('vanzari_butasi_items').insert(itemsPayload)
  399: 
  400:   if (itemsError) {
  401:     await supabase.from('vanzari_butasi').delete().eq('id', comandaId)
  402:     throw toReadableError(itemsError, 'Nu am putut salva produsele comenzii de butasi.')
  403:   }
  404: 
  405:   return getVanzareButasiById(comandaId)
  406: }

  408: export async function updateVanzareButasi(id: string, input: UpdateVanzareButasiInput): Promise<VanzareButasi> {
  409:   const supabase = getSupabase()
  410: 
  411:   const { data: existing, error: existingError } = await supabase
  412:     .from('vanzari_butasi')
  413:     .select('id,status,total_lei,avans_suma,cantitate_butasi,tenant_id')
  414:     .eq('id', id)
  415:     .single()
  416: 
  417:   if (existingError) throw toReadableError(existingError, 'Nu am putut incarca comanda pentru editare.')
  418: 
  419:   const currentStatus = assertStatus(existing.status)
  420: 
  421:   if (currentStatus === 'anulata' && Array.isArray(input.items)) {
  422:     throw new Error('Produsele nu pot fi editate pentru comenzile anulate')
  423:   }
  424: 
  425:   const status = input.status ? assertStatus(input.status) : currentStatus
  426:   const hasItemsUpdate = Array.isArray(input.items)
  427: 
  428:   let computedTotal = Number(existing.total_lei)
  429:   let computedCantitate = Number(existing.cantitate_butasi)
  430:   let firstItemSoi: string | null = null
  431:   let computedPretUnitar = computedCantitate > 0 ? roundTo2(computedTotal / computedCantitate) : 0
  432:   let normalizedItems: NormalizedItemsResult | null = null
  433: 
  434:   if (hasItemsUpdate) {
  435:     normalizedItems = normalizeItems(input.items as VanzareButasiItemInput[])
  436:     computedTotal = normalizedItems.totalLei
  437:     computedCantitate = normalizedItems.totalCantitate
  438:     firstItemSoi = normalizedItems.items[0]?.soi ?? null
  439:     computedPretUnitar = roundTo2(computedTotal / computedCantitate)
  440:   }
  441: 
  442:   const avansSuma = input.avans_suma !== undefined ? roundTo2(Number(input.avans_suma)) : Number(existing.avans_suma)
  443: 
  444:   if (!Number.isFinite(avansSuma) || avansSuma < 0) {
  445:     throw new Error('Avansul trebuie sa fie un numar pozitiv')
  446:   }
  447: 
  448:   const payload: Record<string, unknown> = {
  449:     updated_at: new Date().toISOString(),
  450:     status,
  451:     total_lei: computedTotal,
  452:     avans_suma: avansSuma,
  453:   }
  454: 
  455:   if (input.data_comanda !== undefined) {
  456:     payload.data = input.data_comanda
  457:     payload.data_comanda = input.data_comanda
  458:   }
  459:   if (input.data_livrare_estimata !== undefined) payload.data_livrare_estimata = input.data_livrare_estimata || null
  460:   if (input.client_id !== undefined) payload.client_id = input.client_id || null
  461:   if (input.parcela_sursa_id !== undefined) payload.parcela_sursa_id = input.parcela_sursa_id || null
  462:   if (input.adresa_livrare !== undefined) payload.adresa_livrare = input.adresa_livrare?.trim() || null
  463:   if (input.observatii !== undefined) payload.observatii = input.observatii?.trim() || null
  464:   if (input.avans_data !== undefined) payload.avans_data = input.avans_data || null
  465: 
  466:   if (hasItemsUpdate) {
  467:     payload.soi_butasi = firstItemSoi
  468:     payload.cantitate_butasi = computedCantitate
  469:     payload.pret_unitar_lei = computedPretUnitar
  470:   }
  471: 
  472:   const { error: updateError } = await supabase
  473:     .from('vanzari_butasi')
  474:     .update(payload)
  475:     .eq('id', id)
  476: 
  477:   if (updateError) throw toReadableError(updateError, 'Nu am putut actualiza comanda de butasi.')
  478: 
  479:   if (hasItemsUpdate && normalizedItems) {
  480:     const { error: deleteError } = await supabase
  481:       .from('vanzari_butasi_items')
  482:       .delete()
  483:       .eq('comanda_id', id)
  484: 
  485:     if (deleteError) throw toReadableError(deleteError, 'Nu am putut actualiza liniile comenzii de butasi.')
  486: 
  487:     const itemRows = normalizedItems.items.map((item) => ({
  488:       comanda_id: id,
  489:       soi: item.soi,
  490:       cantitate: item.cantitate,
  491:       pret_unitar: item.pret_unitar,
  492:       subtotal: item.subtotal,
  493:     }))
  494: 
  495:     const { error: insertError } = await supabase
  496:       .from('vanzari_butasi_items')
  497:       .insert(itemRows)
  498: 
  499:     if (insertError) throw toReadableError(insertError, 'Nu am putut salva liniile comenzii de butasi.')
  500:   }
  501: 
  502:   return getVanzareButasiById(id)
  503: }
