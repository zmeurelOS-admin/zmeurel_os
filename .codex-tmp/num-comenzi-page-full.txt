  158: function matchFilter(comanda: Comanda, filter: QuickFilter, today: string): boolean {
  159:   const delivery = comanda.data_livrare ?? ''
  160: 
  161:   if (filter === 'toate') return true
  162:   if (filter === 'livrate') return comanda.status === 'livrata'
  163:   if (filter === 'anulate') return comanda.status === 'anulata'
  164:   if (filter === 'azi') return delivery === today && comanda.status !== 'livrata' && comanda.status !== 'anulata'
  165:   if (filter === 'viitoare') return delivery > today && comanda.status !== 'livrata' && comanda.status !== 'anulata'
  166:   if (filter === 'restante') {
  167:     return Boolean(delivery) && delivery < today && comanda.status !== 'livrata' && comanda.status !== 'anulata'
  168:   }
  169:   return true
  170: }
