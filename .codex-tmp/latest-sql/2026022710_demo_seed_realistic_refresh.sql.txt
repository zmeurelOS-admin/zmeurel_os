  227: create or replace function public.delete_demo_for_tenant(
  228:   p_tenant_id uuid
  229: )
  230: returns jsonb
  231: language plpgsql
  232: security definer
  233: set search_path = public
  234: as $$
  235: declare
  236:   v_user_id uuid := auth.uid();
  237:   v_seed_id uuid;
  238:   v_deleted bigint := 0;
  239:   v_count bigint := 0;
  240:   tbl text;
  241: begin
  242:   if v_user_id is null then
  243:     raise exception 'UNAUTHORIZED';
  244:   end if;
  245: 
  246:   if p_tenant_id is null then
  247:     raise exception 'TENANT_REQUIRED';
  248:   end if;
  249: 
  250:   if not public.user_can_manage_tenant(p_tenant_id, v_user_id) then
  251:     raise exception 'FORBIDDEN';
  252:   end if;
  253: 
  254:   select t.demo_seed_id
  255:   into v_seed_id
  256:   from public.tenants t
  257:   where t.id = p_tenant_id
  258:   for update;
  259: 
  260:   foreach tbl in array array[
  261:     'comenzi',
  262:     'vanzari',
  263:     'vanzari_butasi_items',
  264:     'vanzari_butasi',
  265:     'recoltari',
  266:     'cheltuieli_diverse',
  267:     'activitati_agricole',
  268:     'culegatori',
  269:     'clienti',
  270:     'parcele',
  271:     'investitii',
  272:     'miscari_stoc'
  273:   ] loop
  274:     if to_regclass('public.' || tbl) is not null then
  275:       execute format(
  276:         'with deleted_rows as (
  277:            delete from public.%I
  278:            where tenant_id = $1
  279:              and (
  280:                data_origin = ''demo''
  281:                or ($2 is not null and demo_seed_id = $2)
  282:              )
  283:            returning 1
  284:          )
  285:          select count(*) from deleted_rows',
  286:         tbl
  287:       )
  288:       into v_count
  289:       using p_tenant_id, v_seed_id;
  290: 
  291:       v_deleted := v_deleted + coalesce(v_count, 0);
  292:     end if;
  293:   end loop;
  294: 
  295:   update public.tenants
  296:   set
  297:     demo_seeded = false,
  298:     demo_seed_id = null,
  299:     demo_seeded_at = null,
  300:     updated_at = now()
  301:   where id = p_tenant_id;
  302: 
  303:   return jsonb_build_object(
  304:     'status', 'deleted',
  305:     'tenant_id', p_tenant_id,
  306:     'deleted_rows', v_deleted
  307:   );
  308: end
  309: $$;
